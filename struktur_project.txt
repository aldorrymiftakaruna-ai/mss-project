|           finding_code

+-- BotRegistration.php — Pendaftaran teknisi via Telegram
|   Fields: telegram_id, telegram_username, name, nik,

|           processed_by, processed_at
|   Relasi: processedBy()
|
+-- BotUnknownAsset.php — Asset tidak dikenal dari laporan bot
|   Fields: report_id, keyword_mentioned
|   Relasi: maintenanceReport()
|

|   Fields: name, provider_type, api_key_encrypted, model,

|           request_count_24h, last_used_at, last_health_check,
|           status (healthy/exhausted/error/disabled)
|   Accessor: api_key (decrypt otomatis)
|   Scope: healthy(), byPriority()
|
+-- AiAlias.php         — Alias untuk pencocokan equipment oleh AI
|   Fields: alias_text, asset_id, employee_id, source,
|           confidence, usage_count, status

|   Relasi: asset(), employee(), confirmer()
|

    Fields: provider_id, report_id, tokens_used, request_type,
            response_time_ms, status (success/error), error_message
    Relasi: provider(), report()
    Scope: lastHours(), thisMonth(), success(), error()
    Static: statsByRequestType(), dailyTokensPerProvider(),
            statsPerProvider24h()
